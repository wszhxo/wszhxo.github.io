# 责任链模式

一个请求的处理需要多个对象东昂中的一个或者几个协作处理,就像流水线工厂每个人只做的事情

请求的发送者和接受者解耦

可以随意组合处理顺序

符合开闭原则和单一职责原则

```java
package com.company;

public class 责任链模式 {
    public static void main(String[] args) {
        Request request=new Request().loginon(true).ispermits(false).issensitive(true);
        //组成一条处理器链,可随意组合
        RequestHander requestHander=  new RequestHander(new RequestHander2(new RequestHander3(null)));
        requestHander.process(request);
    }
}
class Request{
    private boolean loginon;//是否登录
    private boolean ispermits;//是否授权
    private boolean issensitive;//是否敏感词

    public Request loginon(boolean loginon) {
        this.loginon=loginon;
        return this;
    }
    public Request ispermits(boolean ispermits) {
        this.ispermits=ispermits;
        return this;
    }
    public Request issensitive(boolean issensitive) {
        this.issensitive=issensitive;
        return this;
    }

    public boolean isLoginon() {
        return loginon;
    }

    public boolean isIspermits() {
        return ispermits;
    }

    public boolean isIssensitive() {
        return issensitive;
    }
}

abstract class Hander{
    Hander next;

    public Hander(Hander next) {
        this.next = next;
    }

    public Hander getNext() {
        return next;
    }

    public void setNext(Hander next) {
        this.next = next;
    }

    abstract boolean process(Request request);
}
//登陆处理器
class RequestHander extends Hander{

    public RequestHander(Hander next) {
        super(next);
    }

    @Override
    boolean process(Request request) {
        System.out.println("处理登录中....");
        if (request.isLoginon()){
            System.out.println("登录校验通过@");
            Hander nextHander=getNext();
            if (null==nextHander)return true;

            return nextHander.process(request);//交给下一个处理器
        }
        System.out.println("登录不通过");
        return false;
    }
}
//是否授权处理器
class RequestHander2 extends Hander{

    public RequestHander2(Hander next) {
        super(next);
    }

    @Override
    boolean process(Request request) {
        System.out.println("处理授权逻辑中....");
        if (request.isIspermits()){
            System.out.println("处理授权通过@");
            Hander nextHander=getNext();
            if (null==nextHander)return true;
            return nextHander.process(request);//交给下一个处理器
        }
        System.out.println("授权不通过");
        return false;
    }
}
//其他处理器同理
class RequestHander3 extends Hander{

    public RequestHander3(Hander next) {
        super(next);
    }

    @Override
    boolean process(Request request) {
        System.out.println("敏感词校验中....");
        if (request.isIspermits()){
            System.out.println("敏感词校验通过@");
            Hander nextHander=getNext();
            if (null==nextHander)return true;
            return nextHander.process(request);//交给下一个处理器
        }
        System.out.println("敏感词校验不通过");
        return false;
    }
}

输出内容如下
    
处理登录中....
登录校验通过@
处理授权逻辑中....
授权不通过

```
